    const TOTAL_STEPS = 10;
    const completed   = new Set();

    const dotsContainer = document.getElementById('progress-dots');
    for (let i = 1; i <= TOTAL_STEPS; i++) {
      const dot = document.createElement('a');
      dot.className   = 'p-dot';
      dot.textContent = i;
      dot.href        = `#step-${i}`;
      dot.id          = `dot-${i}`;
      dot.setAttribute('aria-label', `Go to step ${i}`);

    function toggleStep(header) {
      const card   = header.closest('.step-card');
      const isOpen = card.classList.toggle('open');
      header.setAttribute('aria-expanded', isOpen);
      const stepId = parseInt(card.id.replace('step-', ''));
      updateActiveDot(stepId);
    }

    document.querySelectorAll('.step-header').forEach(header => {
      header.addEventListener('keydown', e => {
        if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); toggleStep(header); }
      });
    });

    function markDone(event, stepNum) {
      event.stopPropagation();
      const btn      = event.target;
      const numBadge = document.getElementById(`num-${stepNum}`);
      const dot      = document.getElementById(`dot-${stepNum}`);
      if (completed.has(stepNum)) {
        completed.delete(stepNum); btn.textContent = 'Mark done'; btn.classList.remove('marked');
        numBadge.classList.remove('done'); dot.classList.remove('done');
      } else {
        completed.add(stepNum); btn.textContent = '✓ Done'; btn.classList.add('marked');
        numBadge.classList.add('done'); dot.classList.remove('active'); dot.classList.add('done');
      }
      const pct  = Math.round((completed.size / TOTAL_STEPS) * 100);
      document.getElementById('progress-fill').style.width = pct + '%';
      document.getElementById('progress-bar').setAttribute('aria-valuenow', pct);
    }

    function updateActiveDot(stepNum) {
      document.querySelectorAll('.p-dot').forEach(d => d.classList.remove('active'));
      const dot = document.getElementById(`dot-${stepNum}`);
      if (dot && !completed.has(stepNum)) dot.classList.add('active');
    }

    function copyPrompt(button) {
      const text = button.closest('.prompt-box').querySelector('.prompt-text');
      navigator.clipboard.writeText(text.innerText).then(() => {
        button.textContent = 'Copied!'; button.classList.add('copied');
        setTimeout(() => { button.textContent = 'Copy'; button.classList.remove('copied'); }, 2000);
      }).catch(() => { button.textContent = 'Select manually'; setTimeout(() => { button.textContent = 'Copy'; }, 2500); });
    }
