  <script>
    // ============================================================
    // SETUP: define total steps and build the progress dot row
    // ============================================================
    const TOTAL_STEPS = 10;
    const completed   = new Set(); // tracks which steps are marked done

    // Build the row of numbered progress dots at the top of the page
    const dotsContainer = document.getElementById('progress-dots');
    for (let i = 1; i <= TOTAL_STEPS; i++) {
      const dot = document.createElement('a');
      dot.className   = 'p-dot';
      dot.textContent = i;
      dot.href        = `#step-${i}`;
      dot.id          = `dot-${i}`;
      dot.setAttribute('aria-label', `Go to step ${i}`);
      dotsContainer.appendChild(dot);
    }

    // Mark step 1's dot as active to start
    document.getElementById('dot-1').classList.add('active');


    // ============================================================
    // FUNCTION: toggleStep(header)
    // Opens or closes a step card when its header is clicked.
    // ============================================================
    function toggleStep(header) {
      const card   = header.closest('.step-card');
      const isOpen = card.classList.toggle('open');
      header.setAttribute('aria-expanded', isOpen);

      // Update the active dot - find which step this is
      const stepId = parseInt(card.id.replace('step-', ''));
      updateActiveDot(stepId);
    }

    // Keyboard support: Enter and Space trigger the toggle
    document.querySelectorAll('.step-header').forEach(header => {
      header.addEventListener('keydown', e => {
        if (e.key === 'Enter' || e.key === ' ') {
          e.preventDefault();
          toggleStep(header);
        }
      });
    });


    // ============================================================
    // FUNCTION: markDone(event, stepNum)
    // Marks a step as completed, updating the button, number badge,
    // progress dots, and the progress bar fill.
    // ============================================================
    function markDone(event, stepNum) {
      // Stop the click from also toggling the step open/closed
      event.stopPropagation();

      const btn     = event.target;
      const numBadge = document.getElementById(`num-${stepNum}`);
      const dot      = document.getElementById(`dot-${stepNum}`);

      // Toggle: if already marked, unmark it; otherwise mark it
      if (completed.has(stepNum)) {
        completed.delete(stepNum);
        btn.textContent = 'Mark done';
        btn.classList.remove('marked');
        numBadge.classList.remove('done');
        dot.classList.remove('done');
      } else {
        completed.add(stepNum);
        btn.textContent = '✓ Done';
        btn.classList.add('marked');
        numBadge.classList.add('done');
        dot.classList.remove('active');
        dot.classList.add('done');
      }

      // Recalculate and update the progress bar width
      const pct = Math.round((completed.size / TOTAL_STEPS) * 100);
      const fill  = document.getElementById('progress-fill');
      const track = document.getElementById('progress-bar');
      fill.style.width = pct + '%';
      track.setAttribute('aria-valuenow', pct);
    }


    // ============================================================
    // FUNCTION: updateActiveDot(stepNum)
    // Highlights the dot for the currently open step.
    // ============================================================
    function updateActiveDot(stepNum) {
      document.querySelectorAll('.p-dot').forEach(d => d.classList.remove('active'));
      const dot = document.getElementById(`dot-${stepNum}`);
      if (dot && !completed.has(stepNum)) {
        dot.classList.add('active');
      }
    }


    // ============================================================
    // FUNCTION: copyPrompt(button)
    // Copies the AI prompt text to the clipboard.
    // ============================================================
    function copyPrompt(button) {
      const box  = button.closest('.prompt-box');
      const text = box.querySelector('.prompt-text');
      navigator.clipboard.writeText(text.innerText).then(() => {
        button.textContent = 'Copied!';
        button.classList.add('copied');
        setTimeout(() => {
          button.textContent = 'Copy';
          button.classList.remove('copied');
        }, 2000);
      }).catch(() => {
        button.textContent = 'Select manually';
        setTimeout(() => { button.textContent = 'Copy'; }, 2500);
      });
    }
  </script>
